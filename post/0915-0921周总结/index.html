  1. 组织树的简单递归算法实现
data.forEach(item=>{
  recursiveTraverse(item);
});

const recursiveTraverse = function (node) {
  if (!node || node['children'].length === 0) {
      node['children'] = null;
      return;
  }
  node['children'].forEach(function(item) {
      recursiveTraverse(item);
  });
};
